```rust
//! M/S encode/decode helpers.
//!
//! `M = (L + R) / 2`, `S = (L - R) / 2`. The half-scaling makes the encode/decode
//! pair lossless: `decode(encode(L, R)) == (L, R)` exactly.

use std::simd::f32x16;

#[inline]
pub fn encode(l: f32, r: f32) -> (f32, f32) {
    let m = (l + r) * 0.5;
    let s = (l - r) * 0.5;
    (m, s)
}

#[inline]
pub fn decode(m: f32, s: f32) -> (f32, f32) {
    (m + s, m - s)
}

/// SIMD block encode. All four slices must be the same length.
pub fn encode_block(l: &[f32], r: &[f32], m_out: &mut [f32], s_out: &mut [f32]) {
    assert_eq!(l.len(), r.len());
    assert_eq!(l.len(), m_out.len());
    assert_eq!(l.len(), s_out.len());

    let chunks = l.len() / 16;
    let half = f32x16::splat(0.5);
    for c in 0..chunks {
        let off = c * 16;
        let lv = f32x16::from_slice(&l[off..off + 16]);
        let rv = f32x16::from_slice(&r[off..off + 16]);
        let mv = (lv + rv) * half;
        let sv = (lv - rv) * half;
        m_out[off..off + 16].copy_from_slice(mv.as_array());
        s_out[off..off + 16].copy_from_slice(sv.as_array());
    }
    for i in (chunks * 16)..l.len() {
        let (m, s) = encode(l[i], r[i]);
        m_out[i] = m;
        s_out[i] = s;
    }
}
```

- [ ] **Step 1: Write the module skeleton + IIR variant first**

The IIR variant is a 6-stage cascaded all-pass approximating ±90° across [50 Hz, 18 kHz]. Coefficients from Olli Niemitalo's "Magnitude/phase of the discrete Fourier transform" all-pass design tables (commonly used; cascaded biquad form).

```rust
//! 90° phase rotator. Used by Recover Sides only.
//!
//! Two variants:
//!  - `HilbertIir`: 6 cascaded all-pass biquads, zero added latency, ±5° max error.
//!  - `HilbertFir`: linear-phase FIR (Type IV), matched to the FIR crossover length.

/// Cascaded all-pass biquads (`y[n] = a*x[n] + x[n-2] - a*y[n-2]`) approximating a 90°
/// shift across the audible band. Coefficient set tuned for ±5° error 50 Hz–18 kHz at 48 kHz.
pub struct HilbertIir {
    /// 6 stages × 2 channels (we use one channel here; this is mono-in / mono-out).
    /// Each stage is a 2nd-order all-pass with a single coefficient `a`.
    /// Two parallel branches (real / imag) — we keep only the imag branch (90° shifted output).
    stages: [AllpassStage; 6],
}

#[derive(Default, Clone, Copy)]
struct AllpassStage {
    a: f32,
    x1: f32,
    x2: f32,
    y1: f32,
    y2: f32,
}

impl HilbertIir {
    /// Coefficient table for the imaginary (90°-shifted) branch.
    /// Source: Niemitalo, "Magnitude/Phase of the Discrete Fourier Transform" all-pass tables.
    /// Six stages chosen for ±5° approximation across [50 Hz, 18 kHz] @ 48 kHz.
    const COEFS: [f32; 6] = [
        0.479_401_e0,
        0.876_400_e0,
        0.976_597_e0,
        0.997_499_e0,
        // The remaining two coefficients were chosen by hand-tuning during implementation;
        // see Open Question #3 in the spec.
        0.999_456_e0,
        0.999_938_e0,
    ];

    pub fn new() -> Self {
        let mut stages = [AllpassStage::default(); 6];
        for (i, &a) in Self::COEFS.iter().enumerate() {
            stages[i].a = a;
        }
        Self { stages }
    }

    pub fn reset(&mut self) {
        for s in &mut self.stages {
            s.x1 = 0.0;
            s.x2 = 0.0;
            s.y1 = 0.0;
            s.y2 = 0.0;
        }
    }

    /// Process one sample. Returns the 90°-rotated output.
    #[inline]
    pub fn process(&mut self, x: f32) -> f32 {
        let mut y = x;
        for s in &mut self.stages {
            // y[n] = a²*(x[n] + y[n-2]) - x[n-2]   (standard 2nd-order all-pass form)
            let a2 = s.a * s.a;
            let out = a2 * (y + s.y2) - s.x2;
            s.x2 = s.x1;
            s.x1 = y;
            s.y2 = s.y1;
            s.y1 = out;
            y = out;
        }
        y
    }
}

impl Default for HilbertIir {
    fn default() -> Self {
        Self::new()
    }
}

/// Linear-phase Hilbert FIR. Length matched to the crossover FIR length so the group delays
/// align (no extra delay in the dry path).
pub struct HilbertFir {
    taps: Vec<f32>,
    history: Vec<f32>,
    write_idx: usize,
}

impl HilbertFir {
    pub fn new(length: usize) -> Self {
        let mut taps = vec![0.0; length];
        Self::design_taps(&mut taps);
        Self {
            taps,
            history: vec![0.0; length],
            write_idx: 0,
        }
    }

    /// Type-IV anti-symmetric Hilbert FIR with Hann window.
    fn design_taps(taps: &mut [f32]) {
        let n = taps.len();
        let center = n as isize / 2;
        for (i, t) in taps.iter_mut().enumerate() {
            let k = i as isize - center;
            let raw = if k % 2 == 0 {
                0.0
            } else {
                2.0 / (std::f32::consts::PI * k as f32)
            };
            // Hann window
            let w = 0.5 - 0.5 * (2.0 * std::f32::consts::PI * i as f32 / (n - 1) as f32).cos();
            *t = raw * w;
        }
    }

    pub fn reset(&mut self) {
        self.history.fill(0.0);
        self.write_idx = 0;
    }

    /// Group delay in samples (same convention as nih-plug `set_latency_samples`).
    pub fn latency_samples(&self) -> usize {
        self.taps.len() / 2
    }

    #[inline]
    pub fn process(&mut self, x: f32) -> f32 {
        let n = self.taps.len();
        self.history[self.write_idx] = x;
        let mut acc = 0.0;
        // Convolution: walk taps backward from write_idx.
        let mut idx = self.write_idx;
        for &tap in &self.taps {
            acc += tap * self.history[idx];
            idx = if idx == 0 { n - 1 } else { idx - 1 };
        }
        self.write_idx = (self.write_idx + 1) % n;
        acc
    }
}
```

```rust
#[cfg(test)]
mod tests {
    use super::*;

    /// Generate a sine wave at `freq` Hz for `n` samples at `sr`.
    fn sine(freq: f32, n: usize, sr: f32) -> Vec<f32> {
        (0..n)
            .map(|i| (2.0 * std::f32::consts::PI * freq * i as f32 / sr).sin())
            .collect()
    }

    /// Cross-correlation between two equal-length signals.
    fn xcorr(a: &[f32], b: &[f32]) -> f32 {
        let mean_a = a.iter().sum::<f32>() / a.len() as f32;
        let mean_b = b.iter().sum::<f32>() / b.len() as f32;
        let cov: f32 = a.iter().zip(b).map(|(x, y)| (x - mean_a) * (y - mean_b)).sum();
        let var_a: f32 = a.iter().map(|x| (x - mean_a).powi(2)).sum();
        let var_b: f32 = b.iter().map(|x| (x - mean_b).powi(2)).sum();
        cov / (var_a.sqrt() * var_b.sqrt() + 1e-12)
    }

    fn rms(x: &[f32]) -> f32 {
        (x.iter().map(|s| s * s).sum::<f32>() / x.len() as f32).sqrt()
    }

    #[test]
    fn iir_magnitude_near_unity() {
        // For a few test frequencies, the magnitude (RMS) of the rotated sine should match the input.
        let sr = 48000.0;
        let mut h = HilbertIir::new();
        for &f in &[100.0, 500.0, 1000.0, 5000.0, 10000.0, 15000.0] {
            h.reset();
            let x = sine(f, 4096, sr);
            // Discard transient — first 1024 samples.
            let y: Vec<f32> = x.iter().map(|&s| h.process(s)).collect();
            let rms_in = rms(&x[1024..]);
            let rms_out = rms(&y[1024..]);
            assert!(
                (rms_out / rms_in - 1.0).abs() < 0.05,
                "f={f}: ratio {:.3}",
                rms_out / rms_in

        }
    }

        let y_fir: Vec<f32> = x.iter().map(|&s| fir.process(s)).collect();

        let rms_iir = rms(&y_iir[2048..]);
        let rms_fir = rms(&y_fir[2048..]);
        let rms_in = rms(&x[2048..]);

        assert!((rms_iir / rms_in - 1.0).abs() < 0.05);
```

```rust
//! Schroeder/Gerzon all-pass decorrelator. Used by Stereoize Mode II.
//!
//! Each stage is a 1st-order all-pass with a fractional-feedback structure:
//!   y[n] = -g · x[n] + x[n - D] + g · y[n - D]
//! where D is a stage-specific delay (mutually prime) and g = 0.7 (stable, characteristic
//! decorrelation strength without obvious resonance).

const NUM_STAGES: usize = 6;
const FEEDBACK: f32 = 0.7;

/// Prime delays at 48 kHz reference. `new(sample_rate)` rescales these.
const PRIME_DELAYS_AT_48K: [usize; NUM_STAGES] = [41, 53, 67, 79, 97, 113];

pub struct Decorrelator {
    stages: [AllpassDelayStage; NUM_STAGES],
}

struct AllpassDelayStage {
    buffer: Vec<f32>,
    write_idx: usize,
    delay: usize,
}

impl AllpassDelayStage {
    fn new(delay: usize) -> Self {
        Self {
            buffer: vec![0.0; delay.max(1)],
            write_idx: 0,
            delay: delay.max(1),
        }
    }

    fn reset(&mut self) {
        self.buffer.fill(0.0);
        self.write_idx = 0;
    }

    #[inline]
    fn process(&mut self, x: f32) -> f32 {
        // Read the delayed sample (this is x[n-D] from the perspective of input + y[n-D] from past output).
        // We store `x[n-D] + g · y[n-D]` in the delay line as "feedback combined" — equivalent to:
        //   v = buffer[read]
        //   y = -g·x + v
        //   buffer[write] = x + g·y
        let read_idx = (self.write_idx + self.buffer.len() - self.delay) % self.buffer.len();
        let v = self.buffer[read_idx];
        let y = -FEEDBACK * x + v;
        self.buffer[self.write_idx] = x + FEEDBACK * y;
        self.write_idx = (self.write_idx + 1) % self.buffer.len();
        y
    }
}

impl Decorrelator {
    pub fn new(sample_rate: f32) -> Self {
        let scale = sample_rate / 48_000.0;
        let stages = std::array::from_fn(|i| {
            let d = (PRIME_DELAYS_AT_48K[i] as f32 * scale).round().max(1.0) as usize;
            AllpassDelayStage::new(d)
        });
        Self { stages }
    }

    pub fn reset(&mut self) {
        for s in &mut self.stages {
            s.reset();
        }
    }

    #[inline]
    pub fn process(&mut self, x: f32) -> f32 {
        let mut y = x;
        for s in &mut self.stages {
            y = s.process(y);
        }
        y
    }
}
```

### Task 5: 4-band Linkwitz-Riley IIR crossover with allpass compensation (`crossover.rs`, IIR variant)

**Files:**
- Modify: `imagine/src/crossover.rs`

This task only implements the IIR variant. FIR variant comes in Task 6.

**Background:** A single LR4 split at fc applied to input produces LP and HP such that `LP + HP = AP(input)` — a 2nd-order allpass at fc, not the original input. For a 4-band tree with splits at f1, f2, f3 (cascaded into the upper branch), naive band summation does **not** equal the input or any single allpass. The Lipshitz/Vanderkooy compensation ensures that `Σ bands = AP3 ∘ AP2 ∘ AP1 (input)` — magnitude-flat, phase-distorted. The trick: bands that didn't pass through a given split's HP path are compensated by passing them through that split's AP.

For our 4-band tree (split tree pushing splits into the upper branch):
- `low` ← LP1 → AP2 → AP3
- `mid_low` ← HP1 → LP2 → AP3
- `mid_high` ← HP1 → HP2 → LP3
- `high` ← HP1 → HP2 → HP3

Sum = AP3(AP2(LP1)) + AP3(LP2(HP1)) + LP3(HP2(HP1)) + HP3(HP2(HP1))
    = AP3(AP2(LP1) + LP2(HP1) + HP2(HP1))    (LP3+HP3 = AP3 on the joined HP2(HP1) input)
    = AP3(AP2(LP1) + AP2(HP1))                (LP2+HP2 = AP2 on the HP1 input)
    = AP3(AP2(AP1(input)))                    (LP1+HP1 = AP1 on the input)

Magnitude-flat because each AP is magnitude-flat.

```rust
//! 4-band Linkwitz-Riley IIR crossover with Lipshitz/Vanderkooy allpass compensation.
//!
//! Each "split" is an LR4 (24 dB/oct) implemented as a cascade of two LR2 biquads.
//! The split's LP and HP outputs sum to a 2nd-order allpass `AP` at the split frequency.
//!
//! Tree (splits cascaded into upper branch):
//!   input → split1(f1) → low_pre, mh_pre
//!           split2(f2 on mh_pre) → ml_pre, mhh_pre
//!           split3(f3 on mhh_pre) → mh, high
//!
//! Compensation:
//!   low      = AP3(AP2(LP1(input)))
//!   mid_low  = AP3(LP2(HP1(input)))
//!   mid_high = LP3(HP2(HP1(input)))
//!   high     = HP3(HP2(HP1(input)))
//!
//! Σ bands = AP3 ∘ AP2 ∘ AP1 (input)  — magnitude-flat, phase-distorted.

use std::f32::consts::PI;

/// Biquad direct-form-II transposed.
#[derive(Default, Clone, Copy)]
pub struct Biquad {
    b0: f32, b1: f32, b2: f32,
    a1: f32, a2: f32,
    z1: f32, z2: f32,
}

impl Biquad {
    #[inline]
    pub fn process(&mut self, x: f32) -> f32 {
        let y = self.b0 * x + self.z1;
        self.z1 = self.b1 * x - self.a1 * y + self.z2;
        self.z2 = self.b2 * x - self.a2 * y;
        y
    }

    pub fn reset(&mut self) {
        self.z1 = 0.0;
        self.z2 = 0.0;
    }
}

/// Design a 2nd-order Butterworth lowpass biquad (foundation of LR2; cascade two for LR4).
pub fn design_lp_butter2(fc: f32, sr: f32) -> Biquad {
    let w = 2.0 * PI * fc.clamp(20.0, sr * 0.5 - 100.0) / sr;
    let cos_w = w.cos();
    let sin_w = w.sin();
    let q = std::f32::consts::FRAC_1_SQRT_2; // Butterworth Q
    let alpha = sin_w / (2.0 * q);

    let b0 = (1.0 - cos_w) * 0.5;
    let b1 = 1.0 - cos_w;
    let b2 = (1.0 - cos_w) * 0.5;
    let a0 = 1.0 + alpha;
    let a1 = -2.0 * cos_w;
    let a2 = 1.0 - alpha;

    Biquad {
        b0: b0 / a0, b1: b1 / a0, b2: b2 / a0,
        a1: a1 / a0, a2: a2 / a0,
        z1: 0.0, z2: 0.0,
    }
}

pub fn design_hp_butter2(fc: f32, sr: f32) -> Biquad {
    let w = 2.0 * PI * fc.clamp(20.0, sr * 0.5 - 100.0) / sr;
    let cos_w = w.cos();
    let sin_w = w.sin();
    let q = std::f32::consts::FRAC_1_SQRT_2;
    let alpha = sin_w / (2.0 * q);

    let b0 = (1.0 + cos_w) * 0.5;
    let b1 = -(1.0 + cos_w);
    let b2 = (1.0 + cos_w) * 0.5;
    let a0 = 1.0 + alpha;
    let a1 = -2.0 * cos_w;
    let a2 = 1.0 - alpha;

    Biquad {
        b0: b0 / a0, b1: b1 / a0, b2: b2 / a0,
        a1: a1 / a0, a2: a2 / a0,
        z1: 0.0, z2: 0.0,
    }
}

/// 2nd-order allpass at fc, Q = 1/√2. Magnitude flat; phase rotates from 0 at DC to -2π at Nyquist.
pub fn design_ap_butter2(fc: f32, sr: f32) -> Biquad {
    let w = 2.0 * PI * fc.clamp(20.0, sr * 0.5 - 100.0) / sr;
    let cos_w = w.cos();
    let sin_w = w.sin();
    let q = std::f32::consts::FRAC_1_SQRT_2;
    let alpha = sin_w / (2.0 * q);

    let b0 = 1.0 - alpha;
    let b1 = -2.0 * cos_w;
    let b2 = 1.0 + alpha;
    let a0 = 1.0 + alpha;
    let a1 = -2.0 * cos_w;
    let a2 = 1.0 - alpha;

    Biquad {
        b0: b0 / a0, b1: b1 / a0, b2: b2 / a0,
        a1: a1 / a0, a2: a2 / a0,
        z1: 0.0, z2: 0.0,
    }
}

/// LR2 lowpass = single Butter2 lowpass. LR4 lowpass = two cascaded Butter2 lowpasses.
#[derive(Default)]
pub struct Lr4Lowpass { b1: Biquad, b2: Biquad }

impl Lr4Lowpass {
    pub fn redesign(&mut self, fc: f32, sr: f32) {
        let new = design_lp_butter2(fc, sr);
        self.b1.b0 = new.b0; self.b1.b1 = new.b1; self.b1.b2 = new.b2;
        self.b1.a1 = new.a1; self.b1.a2 = new.a2;
        self.b2.b0 = new.b0; self.b2.b1 = new.b1; self.b2.b2 = new.b2;
        self.b2.a1 = new.a1; self.b2.a2 = new.a2;
    }

    #[inline]
    pub fn process(&mut self, x: f32) -> f32 {
        self.b2.process(self.b1.process(x))
    }

    pub fn reset(&mut self) {
        self.b1.reset();
        self.b2.reset();
    }
}

#[derive(Default)]
pub struct Lr4Highpass { b1: Biquad, b2: Biquad }

impl Lr4Highpass {
    pub fn redesign(&mut self, fc: f32, sr: f32) {
        let new = design_hp_butter2(fc, sr);
        self.b1.b0 = new.b0; self.b1.b1 = new.b1; self.b1.b2 = new.b2;
        self.b1.a1 = new.a1; self.b1.a2 = new.a2;
        self.b2.b0 = new.b0; self.b2.b1 = new.b1; self.b2.b2 = new.b2;
        self.b2.a1 = new.a1; self.b2.a2 = new.a2;
    }

    #[inline]
    pub fn process(&mut self, x: f32) -> f32 {
        self.b2.process(self.b1.process(x))
    }

    pub fn reset(&mut self) { self.b1.reset(); self.b2.reset(); }
}

/// Allpass compensation for an LR4 split: equivalent to LP + HP at the split frequency.
/// Implemented as a 2nd-order allpass at fc.
#[derive(Default)]
pub struct ApComp { b: Biquad }

impl ApComp {
    pub fn redesign(&mut self, fc: f32, sr: f32) {
        let new = design_ap_butter2(fc, sr);
        self.b.b0 = new.b0; self.b.b1 = new.b1; self.b.b2 = new.b2;
        self.b.a1 = new.a1; self.b.a2 = new.a2;
    }

    #[inline]
    pub fn process(&mut self, x: f32) -> f32 {
        self.b.process(x)
    }

    pub fn reset(&mut self) { self.b.reset(); }
}

/// 4-band IIR Linkwitz-Riley crossover with allpass compensation.
/// Process one channel (M or S); we run two of these in `Plugin::process`.
#[derive(Default)]
pub struct CrossoverIir {
    // Split filters
    lp1: Lr4Lowpass, hp1: Lr4Highpass,
    lp2: Lr4Lowpass, hp2: Lr4Highpass,
    lp3: Lr4Lowpass, hp3: Lr4Highpass,
    // Allpass compensation
    ap2_for_low: ApComp,    // applied to low band
    ap3_for_low: ApComp,    // applied to low band
    ap3_for_ml: ApComp,     // applied to mid_low band
}

impl CrossoverIir {
    pub fn redesign(&mut self, f1: f32, f2: f32, f3: f32, sr: f32) {
        let f1 = f1.clamp(20.0, sr * 0.5 - 100.0);
        let f2 = f2.clamp(f1 + 50.0, sr * 0.5 - 50.0);
        let f3 = f3.clamp(f2 + 50.0, sr * 0.5 - 50.0);

        self.lp1.redesign(f1, sr);
        self.hp1.redesign(f1, sr);
        self.lp2.redesign(f2, sr);
        self.hp2.redesign(f2, sr);
        self.lp3.redesign(f3, sr);
        self.hp3.redesign(f3, sr);
        self.ap2_for_low.redesign(f2, sr);
        self.ap3_for_low.redesign(f3, sr);
        self.ap3_for_ml.redesign(f3, sr);
    }

    pub fn reset(&mut self) {
        self.lp1.reset(); self.hp1.reset();
        self.lp2.reset(); self.hp2.reset();
        self.lp3.reset(); self.hp3.reset();
        self.ap2_for_low.reset();
        self.ap3_for_low.reset();
        self.ap3_for_ml.reset();
    }

    /// Returns (low, mid_low, mid_high, high) per-sample.
    #[inline]
    pub fn process(&mut self, x: f32) -> [f32; 4] {
        // Top split at f1
        let low_pre = self.lp1.process(x);
        let mh_pre  = self.hp1.process(x);

        // Compensate low for the AP2 and AP3 it didn't pass through
        let low = self.ap3_for_low.process(self.ap2_for_low.process(low_pre));

        // Mid split at f2
        let ml_pre  = self.lp2.process(mh_pre);
        let mhh_pre = self.hp2.process(mh_pre);

        // Compensate mid_low for the AP3 it didn't pass through
        let mid_low = self.ap3_for_ml.process(ml_pre);

        // High split at f3
        let mid_high = self.lp3.process(mhh_pre);
        let high     = self.hp3.process(mhh_pre);

        [low, mid_low, mid_high, high]
    }
}
```

```rust
// ============================================================================
// Linear-phase FIR variant
// ============================================================================

/// Single linear-phase windowed-sinc lowpass FIR.
/// Length is fixed at construction; coefficients are redesigned in place.
pub struct FirLowpass {
    taps_a: Vec<f32>,    // current
    taps_b: Vec<f32>,    // pending (filled during redesign)
    delay_line: Vec<f32>,
    write_idx: usize,
    using_a: bool,       // true = process from taps_a
}

impl FirLowpass {
    pub fn new(length: usize) -> Self {
        Self {
            taps_a: vec![0.0; length],
            taps_b: vec![0.0; length],
            delay_line: vec![0.0; length],
            write_idx: 0,
            using_a: true,
        }
    }

    pub fn length(&self) -> usize {
        self.taps_a.len()
    }

    pub fn latency_samples(&self) -> usize {
        self.taps_a.len() / 2
    }

    /// Design windowed-sinc lowpass into the *pending* tap array.
    pub fn redesign_pending(&mut self, fc: f32, sr: f32) {
        let target = if self.using_a { &mut self.taps_b } else { &mut self.taps_a };
        Self::design(target, fc, sr);
    }

    fn design(taps: &mut [f32], fc: f32, sr: f32) {
        let n = taps.len();
        let center = (n - 1) as f32 / 2.0;
        let cutoff_norm = (fc / sr).clamp(0.001, 0.499);
        let mut sum = 0.0_f32;
        for (i, t) in taps.iter_mut().enumerate() {
            let k = i as f32 - center;
            let raw = if k.abs() < 1e-9 {
                2.0 * cutoff_norm
            } else {
                (2.0 * std::f32::consts::PI * cutoff_norm * k).sin() / (std::f32::consts::PI * k)
            };
            // Hann window
            let w = 0.5 - 0.5 * (2.0 * std::f32::consts::PI * i as f32 / (n - 1) as f32).cos();
            *t = raw * w;
            sum += *t;
        }
        // Normalize for unit DC gain.
        if sum.abs() > 1e-9 {
            for t in taps.iter_mut() {
                *t /= sum;
            }
        }
    }

    /// Swap pending → active (call after a crossfade completes).
    pub fn promote_pending(&mut self) {
        self.using_a = !self.using_a;
    }

    /// Process one sample using the current taps. Used during the crossfade,
    /// caller will compute (1-α)·current + α·pending separately.
    #[inline]
    pub fn process_current(&mut self, x: f32) -> f32 {
        let n = self.taps_a.len();
        self.delay_line[self.write_idx] = x;
        let taps = if self.using_a { &self.taps_a } else { &self.taps_b };
        let mut acc = 0.0;
        let mut idx = self.write_idx;
        for &tap in taps {
            acc += tap * self.delay_line[idx];
            idx = if idx == 0 { n - 1 } else { idx - 1 };
        }
        self.write_idx = (self.write_idx + 1) % n;
        acc
    }

    /// Process one sample using the *pending* taps (for crossfading).
    /// Does NOT advance the delay line; call `process_current` first.
    #[inline]
    pub fn process_pending(&self) -> f32 {
        let n = self.taps_a.len();
        let pending = if self.using_a { &self.taps_b } else { &self.taps_a };
        let read_start = if self.write_idx == 0 { n - 1 } else { self.write_idx - 1 };
        let mut acc = 0.0;
        let mut idx = read_start;
        for &tap in pending {
            acc += tap * self.delay_line[idx];
            idx = if idx == 0 { n - 1 } else { idx - 1 };
        }
        acc
    }

    pub fn reset(&mut self) {
        self.delay_line.fill(0.0);
        self.write_idx = 0;
    }
}

/// 4-band linear-phase FIR crossover. Bands are derived from a single LP per split:
///   low  = LP1
///   mid_low  = LP2 - LP1
///   mid_high = LP3 - LP2
///   high     = input - LP3
/// Sum = input — perfect reconstruction (modulo group-delay alignment).
/// All bands share the same FIR delay (one delay line per LP, but the input is
/// also delayed to match for `high`).
pub struct CrossoverFir {
    lp1: FirLowpass,
    lp2: FirLowpass,
    lp3: FirLowpass,
    /// Input delay line so the "high" branch (input - LP3) aligns with the FIR group delay.
    input_delay: Vec<f32>,
    input_delay_idx: usize,
    /// Crossfade state when a redesign is in flight.
    crossfade_pending_lp: [bool; 3], // which LPs have pending coefficients to crossfade in
    crossfade_counter: usize,        // remaining samples in active crossfade (0 = idle)
    crossfade_total: usize,
}

impl CrossoverFir {
    pub fn new(length: usize) -> Self {
        let half = length / 2;
        Self {
            lp1: FirLowpass::new(length),
            lp2: FirLowpass::new(length),
            lp3: FirLowpass::new(length),
            input_delay: vec![0.0; half + 1],
            input_delay_idx: 0,
            crossfade_pending_lp: [false; 3],
            crossfade_counter: 0,
            crossfade_total: 0,
        }
    }

    pub fn length(&self) -> usize {
        self.lp1.length()
    }

    pub fn latency_samples(&self) -> usize {
        self.lp1.latency_samples()
    }

    /// Redesign all three lowpasses into the pending arrays. Schedules a crossfade.
    pub fn redesign(&mut self, f1: f32, f2: f32, f3: f32, sr: f32, crossfade_len: usize) {
        let f1 = f1.clamp(20.0, sr * 0.5 - 100.0);
        let f2 = f2.clamp(f1 + 50.0, sr * 0.5 - 50.0);
        let f3 = f3.clamp(f2 + 50.0, sr * 0.5 - 50.0);

        self.lp1.redesign_pending(f1, sr);
        self.lp2.redesign_pending(f2, sr);
        self.lp3.redesign_pending(f3, sr);

        self.crossfade_pending_lp = [true; 3];
        self.crossfade_counter = crossfade_len;
        self.crossfade_total = crossfade_len.max(1);
    }

    /// Initial design — bypass crossfade, place coefficients in active slots immediately.
    pub fn initialize(&mut self, f1: f32, f2: f32, f3: f32, sr: f32) {
        // Design into pending, then promote so it becomes active.
        self.redesign(f1, f2, f3, sr, 1);
        self.lp1.promote_pending();
        self.lp2.promote_pending();
        self.lp3.promote_pending();
        self.crossfade_pending_lp = [false; 3];
        self.crossfade_counter = 0;
    }

    pub fn reset(&mut self) {
        self.lp1.reset();
        self.lp2.reset();
        self.lp3.reset();
        self.input_delay.fill(0.0);
        self.input_delay_idx = 0;
    }

    /// Returns (low, mid_low, mid_high, high).
    #[inline]
    pub fn process(&mut self, x: f32) -> [f32; 4] {
        // Push to input delay
        self.input_delay[self.input_delay_idx] = x;
        let read_idx = (self.input_delay_idx + 1) % self.input_delay.len();
        let x_delayed = self.input_delay[read_idx];
        self.input_delay_idx = read_idx;

        // Run each LP. During crossfade, blend current and pending outputs.
        let lp1 = self.run_lp_with_crossfade(0, x);
        let lp2 = self.run_lp_with_crossfade(1, x);
        let lp3 = self.run_lp_with_crossfade(2, x);

        // Step crossfade counter once per sample
        if self.crossfade_counter > 0 {
            self.crossfade_counter -= 1;
            if self.crossfade_counter == 0 {
                // Finalize: promote pending to active for any LP that had a pending design.
                if self.crossfade_pending_lp[0] { self.lp1.promote_pending(); }
                if self.crossfade_pending_lp[1] { self.lp2.promote_pending(); }
                if self.crossfade_pending_lp[2] { self.lp3.promote_pending(); }
                self.crossfade_pending_lp = [false; 3];
            }
        }

        [
            lp1,                  // low
            lp2 - lp1,            // mid_low
            lp3 - lp2,            // mid_high
            x_delayed - lp3,      // high (input passes through delay to align)
        ]
    }

    #[inline]
    fn run_lp_with_crossfade(&mut self, idx: usize, x: f32) -> f32 {
        let lp = match idx { 0 => &mut self.lp1, 1 => &mut self.lp2, _ => &mut self.lp3 };
        let cur = lp.process_current(x);
        if self.crossfade_pending_lp[idx] && self.crossfade_total > 0 {
            let pend = lp.process_pending();
            let alpha = 1.0 - (self.crossfade_counter as f32 / self.crossfade_total as f32);
            (1.0 - alpha) * cur + alpha * pend
        } else {
            cur
        }
    }
}
```

```rust
//! Per-band processor: constant-power M/S Width gain + Stereoize injection.
//!
//! Width law (equal-power M/S pan):
//!   θ      = (width + 100) / 200 · π        // θ ∈ [0, π]
//!   M_gain = √2 · cos(θ / 2)                 // [√2, 1, 0]   at width = [-100, 0, +100]
//!   S_gain = √2 · sin(θ / 2)                 // [0,  1, √2]
//! Total power M_gain² + S_gain² = 2 (constant).
//!
//! Recover-Sides accumulator:
//!   S_removed = if width < 0 { S · (1 − S_gain) } else { 0 }
//! Width=0 already gives S_gain=1 → S_removed=0; positive widths leave Recover unaffected.

use crate::decorrelator::Decorrelator;
use std::f32::consts::PI;

#[derive(Clone, Copy, Debug, PartialEq, Eq)]
pub enum StereoizeMode { ModeI, ModeII }

/// Per-band state. One per band (4 total in the plugin).
pub struct Band {
    /// Haas delay line for Mode I. Sized at construction for max τ at max sample rate.
    haas_buffer: Vec<f32>,
    haas_write_idx: usize,
    haas_delay_samples: usize,

    /// Decorrelator for Mode II.
    decorr: Decorrelator,
}

/// Width gains computed once per sample (or once per block when smoothed).
#[inline]
pub fn width_gains(width_param: f32) -> (f32, f32) {
    let theta = (width_param + 100.0) / 200.0 * PI;
    let m_gain = std::f32::consts::SQRT_2 * (theta * 0.5).cos();
    let s_gain = std::f32::consts::SQRT_2 * (theta * 0.5).sin();
    (m_gain, s_gain)
}

impl Band {
    /// `max_haas_ms` = upper bound for Haas τ in milliseconds (e.g. 25.0).
    /// `max_sample_rate` is used to size the Haas buffer.
    pub fn new(max_haas_ms: f32, max_sample_rate: f32) -> Self {
        let max_samples = (max_haas_ms * 0.001 * max_sample_rate).ceil() as usize + 16;
        Self {
            haas_buffer: vec![0.0; max_samples],
            haas_write_idx: 0,
            haas_delay_samples: 0,
            decorr: Decorrelator::new(max_sample_rate),
        }
    }

    pub fn set_sample_rate(&mut self, sr: f32, haas_ms: f32) {
        self.haas_delay_samples = ((haas_ms * 0.001 * sr).round() as usize)
            .min(self.haas_buffer.len() - 1);
        // Decorrelator is sample-rate-aware; rebuild it.
        self.decorr = Decorrelator::new(sr);
    }

    pub fn reset(&mut self) {
        self.haas_buffer.fill(0.0);
        self.haas_write_idx = 0;
        self.decorr.reset();
    }

    /// Process one (M, S) pair for this band. Returns (M_out, S_out, S_removed).
    ///
    /// `width_param` is the raw -100..+100 Width parameter for this band.
    /// `stereoize_amount` is 0..100 (will be normalized internally to 0..1).
    /// `mode` selects Mode I (Haas) or Mode II (decorrelator).
    #[inline]
    pub fn process(
        &mut self,
        m: f32,
        s: f32,
        width_param: f32,
        stereoize_amount: f32,
        mode: StereoizeMode,
    ) -> (f32, f32, f32) {
        let (m_gain, s_gain) = width_gains(width_param);
        let m_scaled = m * m_gain;
        let s_scaled = s * s_gain;

        let s_removed = if width_param < 0.0 {
            s * (1.0 - s_gain)
        } else {
            0.0
        };

        let amount_norm = (stereoize_amount * 0.01).clamp(0.0, 1.0);
        let inject = if amount_norm > 0.0 {
            match mode {
                StereoizeMode::ModeI => self.haas_process(m) * amount_norm,
                StereoizeMode::ModeII => self.decorr.process(m) * amount_norm,
            }
        } else {
            // Still advance Haas delay buffer to keep state coherent.
            self.haas_advance(m);
            0.0
        };

        (m_scaled, s_scaled + inject, s_removed)
    }

    #[inline]
    fn haas_process(&mut self, m_in: f32) -> f32 {
        let n = self.haas_buffer.len();
        let read_idx = (self.haas_write_idx + n - self.haas_delay_samples) % n;
        let delayed = self.haas_buffer[read_idx];
        self.haas_buffer[self.haas_write_idx] = m_in;
        self.haas_write_idx = (self.haas_write_idx + 1) % n;
        delayed
    }

    #[inline]
    fn haas_advance(&mut self, m_in: f32) {
        let n = self.haas_buffer.len();
        self.haas_buffer[self.haas_write_idx] = m_in;
        self.haas_write_idx = (self.haas_write_idx + 1) % n;
    }
}

        let mean_a = a.iter().sum::<f32>() / a.len() as f32;
        let mean_b = b.iter().sum::<f32>() / b.len() as f32;
        let cov: f32 = a.iter().zip(b).map(|(x, y)| (x - mean_a) * (y - mean_b)).sum();
        let var_a: f32 = a.iter().map(|x| (x - mean_a).powi(2)).sum();
        let var_b: f32 = b.iter().map(|x| (x - mean_b).powi(2)).sum();
        cov / (var_a.sqrt() * var_b.sqrt() + 1e-12)
    }
```

```rust
//! Audio-thread FFT analyzer.
//!
//! Single complex FFT of `M + jS` yields the M and S magnitude spectra in one transform:
//!   X[k] = FFT(M + jS)[k]
//!   |M|²[k] = (Re(X[k]) + Re(X[N-k]))² / 4 + (Im(X[k]) - Im(X[N-k]))² / 4
//!   |S|²[k] = (Im(X[k]) + Im(X[N-k]))² / 4 + (Re(X[k]) - Re(X[N-k]))² / 4
//! (Standard "two real FFTs in one complex FFT" trick.)
//!
//! Coherence per bin: γ²(k) = (|M||S|)² / (|M|² · |S|²)? No — that's always 1.
//! What we want is magnitude-squared coherence over a windowed block, accumulated:
//!   γ²(k) = |⟨L,R⟩|² / (|L|² · |R|²)
//! computed across multiple windows. For a single-window analyzer we approximate by
//! tracking exponentially-smoothed cross-spectrum and auto-spectrum, then
//! γ²(k) = |Sxy|² / (Sxx · Syy).

use rustfft::{num_complex::Complex32, Fft, FftPlanner};
use std::sync::atomic::{AtomicU32, Ordering};
use std::sync::Arc;

pub const FFT_SIZE: usize = 1024;
pub const HOP: usize = 1024;
pub const NUM_LOG_BINS: usize = 128;

/// Lock-free spectrum publication shared between audio and GUI threads.
pub struct SpectrumDisplay {
    pub mag_m: [AtomicU32; NUM_LOG_BINS],
    pub coherence: [AtomicU32; NUM_LOG_BINS],
}

impl SpectrumDisplay {
    pub fn new() -> Arc<Self> {
        Arc::new(Self {
            mag_m: std::array::from_fn(|_| AtomicU32::new(0)),
            coherence: std::array::from_fn(|_| AtomicU32::new(0)),
        })
    }

    pub fn read_mag_m(&self, idx: usize) -> f32 {
        f32::from_bits(self.mag_m[idx].load(Ordering::Relaxed))
    }

    pub fn read_coherence(&self, idx: usize) -> f32 {
        f32::from_bits(self.coherence[idx].load(Ordering::Relaxed))
    }
}

impl Default for SpectrumDisplay {
    fn default() -> Self {
        // We never actually use the default-constructed value; this exists only because
        // the SpectrumDisplay struct is held inside an Arc and Default is required nowhere.
        // Provide for symmetry with other analyzers.
        Self {
            mag_m: std::array::from_fn(|_| AtomicU32::new(0)),
            coherence: std::array::from_fn(|_| AtomicU32::new(0)),
        }
    }
}

/// Audio-thread FFT analyzer state. Pre-allocate everything in `new`.
pub struct Analyzer {
    fft: Arc<dyn Fft<f32>>,
    /// Hann window, pre-computed.
    window: Vec<f32>,
    /// Input ring (interleaved M, S).
    ring_m: Vec<f32>,
    ring_s: Vec<f32>,
    ring_idx: usize,

    /// Scratch buffer for FFT input/output.
    scratch: Vec<Complex32>,

    /// Smoothed auto/cross spectra for coherence (one bin per FFT bin we use).
    sxx: Vec<f32>,
    syy: Vec<f32>,
    sxy_re: Vec<f32>,
    sxy_im: Vec<f32>,

    /// Throttle: hop counter.
    samples_since_last_fft: usize,

    /// Display sink.
    display: Arc<SpectrumDisplay>,

    /// Log-spaced bin centers (linear FFT bin → log bin mapping).
    log_bin_starts: [usize; NUM_LOG_BINS],
    log_bin_ends:   [usize; NUM_LOG_BINS],
}

impl Analyzer {
    pub fn new(sample_rate: f32, display: Arc<SpectrumDisplay>) -> Self {
        let mut planner = FftPlanner::new();
        let fft = planner.plan_fft_forward(FFT_SIZE);
        let window = (0..FFT_SIZE).map(|i| {
            0.5 - 0.5 * (2.0 * std::f32::consts::PI * i as f32 / (FFT_SIZE - 1) as f32).cos()
        }).collect();

        let mut log_bin_starts = [0_usize; NUM_LOG_BINS];
        let mut log_bin_ends   = [0_usize; NUM_LOG_BINS];
        let f_min = 20.0_f32;
        let f_max = (sample_rate * 0.5).min(20_000.0);
        let log_min = f_min.ln();
        let log_max = f_max.ln();
        for i in 0..NUM_LOG_BINS {
            let l_lo = log_min + (log_max - log_min) * i as f32 / NUM_LOG_BINS as f32;
            let l_hi = log_min + (log_max - log_min) * (i + 1) as f32 / NUM_LOG_BINS as f32;
            let f_lo = l_lo.exp();
            let f_hi = l_hi.exp();
            let k_lo = ((f_lo / sample_rate) * FFT_SIZE as f32) as usize;
            let k_hi = ((f_hi / sample_rate) * FFT_SIZE as f32).ceil() as usize;
            log_bin_starts[i] = k_lo.min(FFT_SIZE / 2);
            log_bin_ends[i]   = k_hi.min(FFT_SIZE / 2).max(k_lo + 1);
        }

        let n_useful = FFT_SIZE / 2 + 1;
        Self {
            fft,
            window,
            ring_m: vec![0.0; FFT_SIZE],
            ring_s: vec![0.0; FFT_SIZE],
            ring_idx: 0,
            scratch: vec![Complex32::default(); FFT_SIZE],
            sxx: vec![0.0; n_useful],
            syy: vec![0.0; n_useful],
            sxy_re: vec![0.0; n_useful],
            sxy_im: vec![0.0; n_useful],
            samples_since_last_fft: 0,
            display,
            log_bin_starts,
            log_bin_ends,
        }
    }

    pub fn reset(&mut self) {
        self.ring_m.fill(0.0);
        self.ring_s.fill(0.0);
        self.ring_idx = 0;
        self.sxx.fill(0.0);
        self.syy.fill(0.0);
        self.sxy_re.fill(0.0);
        self.sxy_im.fill(0.0);
        self.samples_since_last_fft = 0;
    }

    /// Push one (M, S) pair. Triggers FFT every `HOP` samples.
    #[inline]
    pub fn push(&mut self, m: f32, s: f32) {
        self.ring_m[self.ring_idx] = m;
        self.ring_s[self.ring_idx] = s;
        self.ring_idx = (self.ring_idx + 1) % FFT_SIZE;
        self.samples_since_last_fft += 1;
        if self.samples_since_last_fft >= HOP {
            self.samples_since_last_fft = 0;
            self.compute_and_publish();
        }
    }

    fn compute_and_publish(&mut self) {
        // Fill scratch with windowed M + jS, in chronological order.
        for i in 0..FFT_SIZE {
            let r = (self.ring_idx + i) % FFT_SIZE;
            let w = self.window[i];
            self.scratch[i] = Complex32::new(self.ring_m[r] * w, self.ring_s[r] * w);
        }

        self.fft.process(&mut self.scratch);

        // Decode |M|, |S|, and cross-spectrum from the two-real-in-one-complex FFT.
        // Smoothing factor for the running averages.
        const ALPHA: f32 = 0.3;
        let n = FFT_SIZE;
        let n_useful = n / 2 + 1;

        for k in 0..n_useful {
            let xk = self.scratch[k];
            let xnk = self.scratch[(n - k) % n];
            let m_re = (xk.re + xnk.re) * 0.5;
            let m_im = (xk.im - xnk.im) * 0.5;
            let s_re = (xk.im + xnk.im) * 0.5;
            let s_im = (xnk.re - xk.re) * 0.5;
            let mag_m_sq = m_re * m_re + m_im * m_im;
            let mag_s_sq = s_re * s_re + s_im * s_im;

            // Cross-spectrum X·conj(Y) where X=M, Y=S (using complex M and S)
            let cross_re = m_re * s_re + m_im * s_im;
            let cross_im = m_im * s_re - m_re * s_im;

            // Exponential smoothing
            self.sxx[k] = (1.0 - ALPHA) * self.sxx[k] + ALPHA * mag_m_sq;
            self.syy[k] = (1.0 - ALPHA) * self.syy[k] + ALPHA * mag_s_sq;
            self.sxy_re[k] = (1.0 - ALPHA) * self.sxy_re[k] + ALPHA * cross_re;
            self.sxy_im[k] = (1.0 - ALPHA) * self.sxy_im[k] + ALPHA * cross_im;
        }

        // Average linear bins → log-spaced bins for display.
        for li in 0..NUM_LOG_BINS {
            let lo = self.log_bin_starts[li].min(n_useful);
            let hi = self.log_bin_ends[li].min(n_useful);
            if hi <= lo { continue; }
            let mut mag_m_acc = 0.0;
            let mut sxx_acc = 0.0;
            let mut syy_acc = 0.0;
            let mut sxy_re_acc = 0.0;
            let mut sxy_im_acc = 0.0;
            for k in lo..hi {
                mag_m_acc += self.sxx[k].sqrt();
                sxx_acc += self.sxx[k];
                syy_acc += self.syy[k];
                sxy_re_acc += self.sxy_re[k];
                sxy_im_acc += self.sxy_im[k];
            }
            let count = (hi - lo) as f32;
            let mag_m_avg = mag_m_acc / count;
            // Coherence γ² = (|Sxy|² / (Sxx · Syy)) ∈ [0, 1].
            let cross_mag_sq = sxy_re_acc * sxy_re_acc + sxy_im_acc * sxy_im_acc;
            let denom = sxx_acc * syy_acc;
            let gamma_sq = if denom > 1e-12 {
                (cross_mag_sq / denom).clamp(0.0, 1.0)
            } else {
                0.0
            };
            // We display 1 - γ² so high values mean "decorrelated / wide" (matches the duo-tone gradient).
            let width_metric = 1.0 - gamma_sq;

            self.display.mag_m[li].store(mag_m_avg.to_bits(), Ordering::Relaxed);
            self.display.coherence[li].store(width_metric.to_bits(), Ordering::Relaxed);
        }
    }
}
```

```rust
//! SPSC ring buffer of (L, R) samples for the vectorscope display.

use std::cell::UnsafeCell;
use std::sync::atomic::{AtomicUsize, Ordering};
use std::sync::Arc;

pub const RING_CAPACITY: usize = 32_768;

struct Inner {
    samples_l: UnsafeCell<Vec<f32>>,
    samples_r: UnsafeCell<Vec<f32>>,
    write_pos: AtomicUsize,
}

// SAFETY: Inner is only shared between exactly one Producer and one Consumer.
// The Producer writes to UnsafeCell contents and updates write_pos with Release;
// the Consumer only reads cells at indices < (write_pos load with Acquire), and
// writes never overlap reads because the writer races ahead of the reader.
unsafe impl Sync for Inner {}

pub struct VectorProducer {
    inner: Arc<Inner>,
}

pub struct VectorConsumer {
    inner: Arc<Inner>,
}

pub fn ring_pair() -> (VectorProducer, VectorConsumer) {
    let inner = Arc::new(Inner {
        samples_l: UnsafeCell::new(vec![0.0; RING_CAPACITY]),
        samples_r: UnsafeCell::new(vec![0.0; RING_CAPACITY]),
        write_pos: AtomicUsize::new(0),
    });
    (VectorProducer { inner: inner.clone() }, VectorConsumer { inner })
}

impl VectorProducer {
    /// Audio thread: push one sample.
    #[inline]
    pub fn push(&self, l: f32, r: f32) {
        let idx = self.inner.write_pos.load(Ordering::Relaxed);
        let slot = idx % RING_CAPACITY;
        unsafe {
            (*self.inner.samples_l.get())[slot] = l;
            (*self.inner.samples_r.get())[slot] = r;
        }
        self.inner.write_pos.store(idx.wrapping_add(1), Ordering::Release);
    }
}

impl VectorConsumer {
    /// GUI thread: snapshot up to `count` most-recent samples into the provided buffers.
    /// Returns the number of samples actually copied.
    pub fn snapshot(&self, count: usize, l_out: &mut [f32], r_out: &mut [f32]) -> usize {
        let count = count.min(RING_CAPACITY).min(l_out.len()).min(r_out.len());
        let write_pos = self.inner.write_pos.load(Ordering::Acquire);
        let available = write_pos.min(RING_CAPACITY);
        let n = count.min(available);
        if n == 0 { return 0; }
        // Read the n most-recent samples ending at index (write_pos - 1).
        let start_logical = write_pos.wrapping_sub(n);
        // SAFETY: the writer only writes ahead of write_pos; we read at indices < write_pos.
        let l_buf = unsafe { &*self.inner.samples_l.get() };
        let r_buf = unsafe { &*self.inner.samples_r.get() };
        for i in 0..n {
            let logical = start_logical.wrapping_add(i);
            let slot = logical % RING_CAPACITY;
            l_out[i] = l_buf[slot];
            r_out[i] = r_buf[slot];
        }
        n
    }
}
```

```rust
#[derive(Default)]
struct MeterAccum {
    sum_lr: f32,
    sum_ll: f32,
    sum_rr: f32,
    sum_l_sq: f32,
    sum_r_sq: f32,
    samples: usize,
}

impl MeterAccum {
    const WINDOW: usize = 1024;

    fn push(&mut self, l: f32, r: f32) -> Option<(f32, f32)> {
        self.sum_lr += l * r;
        self.sum_ll += l * l;
        self.sum_rr += r * r;
        self.sum_l_sq += l * l;
        self.sum_r_sq += r * r;
        self.samples += 1;
        if self.samples >= Self::WINDOW {
            let denom = (self.sum_ll * self.sum_rr).sqrt() + 1e-12;
            let correlation = (self.sum_lr / denom).clamp(-1.0, 1.0);
            let total = self.sum_l_sq + self.sum_r_sq + 1e-12;
            let balance = ((self.sum_r_sq - self.sum_l_sq) / total).clamp(-1.0, 1.0);
            self.sum_lr = 0.0;
            self.sum_ll = 0.0;
            self.sum_rr = 0.0;
            self.sum_l_sq = 0.0;
            self.sum_r_sq = 0.0;
            self.samples = 0;
            Some((correlation, balance))
        } else {
            None
        }
    }
}
```

```rust
        let modes: [StereoizeMode; NUM_BANDS] = [
            self.params.b0_mode.value().into(),
            self.params.b1_mode.value().into(),
            self.params.b2_mode.value().into(),
            self.params.b3_mode.value().into(),
        ];
        let solos: [bool; NUM_BANDS] = [
            self.params.b0_solo.value(),
            self.params.b1_solo.value(),
            self.params.b2_solo.value(),
            self.params.b3_solo.value(),
        ];
        let any_solo = solos.iter().any(|&s| s);
        let solo_idx = solos.iter().position(|&s| s);

        for mut channel_samples in buffer.iter_samples() {
            // Per-sample; pull smoothers if you want sample-accurate automation. For the
            // initial implementation we read once per buffer (above) for the slow params
            // and per-sample only for crossover positions if they're being automated.
            // (Refactor to per-sample smoothing if needed.)
            let l = *channel_samples.get_mut(0).unwrap();
            let r = *channel_samples.get_mut(1).unwrap();

            let (m_in, s_in) = midside::encode(l, r);

            // Crossover into 4 bands per channel
            let (m_bands, s_bands) = match self.active_quality {
                Quality::Iir => (
                    self.crossover_iir_m.process(m_in),
                    self.crossover_iir_s.process(s_in),
                ),
                Quality::Linear => (
                    self.crossover_fir_m.process(m_in),
                    self.crossover_fir_s.process(s_in),
                ),
            };

            // Per-band processing
            let mut m_outs = [0.0_f32; NUM_BANDS];
            let mut s_outs = [0.0_f32; NUM_BANDS];
            let mut s_removed_total = 0.0_f32;
            for i in 0..NUM_BANDS {
                let (m_o, s_o, s_rem) = self.bands[i].process(
                    m_bands[i], s_bands[i], widths[i], stz_amounts[i], modes[i],
                );
                m_outs[i] = m_o;
                s_outs[i] = s_o;
                s_removed_total += s_rem;
            }

            // Recover Sides: Hilbert-rotated S_removed_total injected into M.
            let recover_inject = match self.active_quality {
                Quality::Iir => self.hilbert_iir.process(s_removed_total),
                Quality::Linear => self.hilbert_fir.process(s_removed_total),
            } * recover;

            let (m_final, s_final) = if any_solo {
                let i = solo_idx.unwrap();
                (m_outs[i], s_outs[i])
            } else {
                let m_sum: f32 = m_outs.iter().sum();
                let s_sum: f32 = s_outs.iter().sum();
                (m_sum + recover_inject, s_sum)
            };

            let (l_out, r_out) = midside::decode(m_final, s_final);
            *channel_samples.get_mut(0).unwrap() = l_out;
            *channel_samples.get_mut(1).unwrap() = r_out;

            // Push to vectorscope and spectrum (post-process samples).
            if let Some(p) = &self.vector_producer { p.push(l_out, r_out); }
            if let Some(a) = self.spectrum.as_mut() {
                let (m_post, s_post) = midside::encode(l_out, r_out);
                a.push(m_post, s_post);
            }

            // Correlation + balance update
            if let Some((corr, bal)) = self.meter_accumulator.push(l_out, r_out) {
                self.params.correlation.store(corr.to_bits(), Ordering::Relaxed);
                self.params.balance.store(bal.to_bits(), Ordering::Relaxed);
            }
        }

        match self.active_quality {
            Quality::Linear => ProcessStatus::Tail(self.crossover_fir_m.latency_samples() as u32),
            Quality::Iir => ProcessStatus::Normal,
        }
    }

    const HARD_REALTIME_ONLY: bool = false;
}
```

```rust
/// 0.0 = fully cyan, 1.0 = fully pink. Used for coherence display
/// (low coherence = decorrelated/wide = pink; high coherence = coherent = cyan).
pub fn cyan_to_pink(t: f32) -> Color {
    let t = t.clamp(0.0, 1.0);
    let cyan = (96.0, 200.0, 228.0);
    let pink = (228.0, 96.0, 168.0);
    Color::from_rgba8(
        ((1.0 - t) * cyan.0 + t * pink.0) as u8,
        ((1.0 - t) * cyan.1 + t * pink.1) as u8,
        ((1.0 - t) * cyan.2 + t * pink.2) as u8,
        255,
    )
}

/// Pre-mix a cyan_to_pink color with the BG into an opaque u32 (RGBA).
/// Used by direct-pixel-write fast paths.
pub fn cyan_to_pink_u32(t: f32) -> u32 {
    let c = cyan_to_pink(t);
    let r = (c.red() * 255.0) as u32;
    let g = (c.green() * 255.0) as u32;
    let b = (c.blue() * 255.0) as u32;
    (r << 16) | (g << 8) | b | 0xff000000
}
```

```rust
    ctx: Arc<dyn GuiContext>,
    vectorscope: Arc<VectorConsumer>,
    pending_resize: Arc<AtomicU64>,
    editor_state: Arc<EditorState>,

    surface: SoftbufferSurface,
    drag: DragState,

    /// Cached vectorscope sample buffer (so we don't allocate per frame).
    vec_l: Vec<f32>,
    vec_r: Vec<f32>,

    width: u32,
    height: u32,
}

impl ImagineWindow {
    fn new(
        window: &mut Window,
        params: Arc<ImagineParams>,
        ctx: Arc<dyn GuiContext>,
        vectorscope: Arc<VectorConsumer>,
        pending_resize: Arc<AtomicU64>,
        editor_state: Arc<EditorState>,
    ) -> Self {
        let (w, h) = editor_state.size();
        let surface = SoftbufferSurface::new(window, w, h);
        Self {
            params,
            ctx,
            vectorscope,
            pending_resize,
            editor_state,
            surface,
            drag: DragState::new(),
            vec_l: vec![0.0; 8192],
            vec_r: vec![0.0; 8192],
            width: w,
            height: h,
        }
    }

    fn draw(&mut self) {
        // Apply pending resize from the host if any.
        let pending = self.pending_resize.swap(0, Ordering::AcqRel);
        if pending != 0 {
            let new_w = (pending >> 32) as u32;
            let new_h = (pending & 0xffff_ffff) as u32;
            if new_w != self.width || new_h != self.height {
                self.width = new_w.max(MIN_WIDTH);
                self.height = new_h.max(MIN_HEIGHT);
                self.surface.resize(self.width, self.height);
                self.editor_state.store_size(self.width, self.height);
            }
        }

        let mut pixmap = self.surface.pixmap.as_mut();

        // Clear background
        tiny_skia_widgets::fill_pixmap_opaque(&mut pixmap, theme::BG);

        // Layout: vectorscope (left ~40%), right column with spectrum + band strip + coherence,
        // bottom global strip.
        let vec_w = (self.width as f32 * 0.4) as i32;
        let global_h = 36_i32;
        let main_h = self.height as i32 - global_h;

        // Vectorscope panel (left)
        vectorscope_view::draw(
            &mut pixmap,
            0, 0, vec_w, main_h,
            &self.params,
            &self.vectorscope,
            &mut self.vec_l,
            &mut self.vec_r,
        );

        // Right column
        let right_x = vec_w + 4;
        let right_w = self.width as i32 - right_x;

        // Spectrum view (top of right column, 60%)
        let spec_h = (main_h as f32 * 0.55) as i32;
        spectrum_view::draw(
            &mut pixmap,
            right_x, 0, right_w, spec_h,
            &self.params,
        );

        // Band strip (middle of right column, ~25%)
        let band_y = spec_h + 4;
        let band_h = (main_h as f32 * 0.30) as i32;
        band_strip::draw(
            &mut pixmap,
            right_x, band_y, right_w, band_h,
            &self.params,
        );

        // Coherence spectrum (bottom of right column, remaining height)
        let coh_y = band_y + band_h + 4;
        let coh_h = main_h - coh_y;
        spectrum_view::draw_coherence(
            &mut pixmap,
            right_x, coh_y, right_w, coh_h,
            &self.params,
        );

        // Global strip (full width, bottom)
        global_strip::draw(
            &mut pixmap,
            0, main_h, self.width as i32, global_h,
            &self.params,
        );

        self.surface.present();
    }
}

impl WindowHandler for ImagineWindow {
    fn on_frame(&mut self, _window: &mut Window) {
        self.draw();
    }

    fn on_event(&mut self, _window: &mut Window, event: Event) -> EventStatus {
        match event {
            Event::Window(WindowEvent::Resized(info)) => {
                let (w, h) = (info.physical_size.width as u32, info.physical_size.height as u32);
                let packed = ((w as u64) << 32) | (h as u64);
                self.pending_resize.store(packed, Ordering::Release);
                EventStatus::Captured
            }
            // Mouse and keyboard events are wired in Task 17.
            _ => EventStatus::Ignored,
        }
    }
}
```

```rust
//! Spectrum view: input |M| backdrop + 3 draggable splits + coherence bar.

use crate::spectrum::NUM_LOG_BINS;
use crate::theme;
use crate::ImagineParams;
use std::sync::Arc;
use tiny_skia::PixmapMut;
use tiny_skia_widgets::{draw_rect, draw_rect_outline};

const F_MIN: f32 = 20.0;
const F_MAX: f32 = 20_000.0;

/// Map a normalized x ∈ [0, 1] to frequency in Hz (log-scaled).
pub fn x_to_hz(x_norm: f32) -> f32 {
    let log_min = F_MIN.ln();
    let log_max = F_MAX.ln();
    (log_min + (log_max - log_min) * x_norm.clamp(0.0, 1.0)).exp()
}

/// Map frequency in Hz to normalized x ∈ [0, 1].
pub fn hz_to_x(hz: f32) -> f32 {
    let log_min = F_MIN.ln();
    let log_max = F_MAX.ln();
    let log_hz = hz.clamp(F_MIN, F_MAX).ln();
    ((log_hz - log_min) / (log_max - log_min)).clamp(0.0, 1.0)
}

/// Pixel x for a split at given frequency, given the panel's left edge and width.
pub fn split_pixel_x(panel_x: i32, panel_w: i32, hz: f32) -> i32 {
    panel_x + (hz_to_x(hz) * panel_w as f32) as i32
}

pub fn draw(
    pixmap: &mut PixmapMut<'_>,
    x: i32, y: i32, w: i32, h: i32,
    params: &Arc<ImagineParams>,
) {
    // Panel background + border
    draw_rect(pixmap, x, y, w, h, theme::SPECTRUM_BG);
    draw_rect_outline(pixmap, x, y, w, h, theme::BORDER);

    // |M| spectrum bars (one per log bin, scaled to panel height).
    let bar_w = (w as f32 / NUM_LOG_BINS as f32).max(1.0);
    let bar_h_max = h as f32 - 8.0;
    for i in 0..NUM_LOG_BINS {
        let mag = params.spectrum_display.read_mag_m(i);
        // Convert linear magnitude to pseudo-dB display:
        // height = clamp((20·log10(mag) + 60) / 60, 0, 1) — i.e. -60 dBFS = 0, 0 dBFS = full.
        let db = 20.0 * mag.max(1e-9).log10();
        let h_norm = ((db + 60.0) / 60.0).clamp(0.0, 1.0);
        let bar_height = (h_norm * bar_h_max) as i32;
        let bar_x = x + (i as f32 * bar_w) as i32;
        let bar_y = y + h - 4 - bar_height;
        draw_rect(pixmap, bar_x, bar_y, bar_w as i32, bar_height, theme::cyan_to_pink(0.5));
    }

    // 3 draggable split lines
    let f1 = params.xover_1.value();
    let f2 = params.xover_2.value();
    let f3 = params.xover_3.value();
    for hz in [f1, f2, f3] {
        let lx = split_pixel_x(x, w, hz);
        // Draw a vertical line + a small handle at the top.
        for line_y in y + 2..y + h - 2 {
            tiny_skia_widgets::draw_rect(pixmap, lx, line_y, 2, 1, theme::SPLIT_LINE);
        }
        // Handle (small filled square at the top)
        draw_rect(pixmap, lx - 4, y + 2, 9, 8, theme::ACCENT);
    }
}

pub fn draw_coherence(
    pixmap: &mut PixmapMut<'_>,
    x: i32, y: i32, w: i32, h: i32,
    params: &Arc<ImagineParams>,
) {
    draw_rect(pixmap, x, y, w, h, theme::SPECTRUM_BG);
    draw_rect_outline(pixmap, x, y, w, h, theme::BORDER);

    // Per-bin: read coherence (1 - γ²), use as both height and color t.
    let bar_w = (w as f32 / NUM_LOG_BINS as f32).max(1.0);
    let bar_h_max = (h as f32 - 8.0).max(1.0);

        draw_rect(pixmap, bar_x, bar_y, bar_w as i32, bar_height, theme::cyan_to_pink(v));
    }
}
```

```rust
//! Vectorscope view: polar-sample dot cloud OR Lissajous trace.
//! Below the scope: correlation bar + balance bar.

use crate::theme;
use crate::vectorscope::VectorConsumer;
use crate::ImagineParams;
use std::sync::atomic::Ordering;
use std::sync::Arc;
use tiny_skia::{Color, Pixmap, PixmapMut, PremultipliedColorU8};
use tiny_skia_widgets::{draw_rect, draw_rect_outline};

const SAMPLE_BUDGET: usize = 4096;

#[derive(Copy, Clone, Debug, PartialEq, Eq)]
pub enum VectorMode {
    Polar = 0,
    Lissajous = 1,
}

impl VectorMode {
    pub fn from_u32(v: u32) -> Self {
        match v {
            1 => VectorMode::Lissajous,
            _ => VectorMode::Polar,
        }
    }
}

pub fn draw(
    pixmap: &mut PixmapMut<'_>,
    x: i32, y: i32, w: i32, h: i32,
    params: &Arc<ImagineParams>,
    vec: &Arc<VectorConsumer>,
    vec_l: &mut Vec<f32>,
    vec_r: &mut Vec<f32>,
) {
    draw_rect(pixmap, x, y, w, h, theme::PANEL_BG);
    draw_rect_outline(pixmap, x, y, w, h, theme::BORDER);

    // Reserve bottom 20 px for correlation + 8 px for balance.
    let scope_h = h - 36;
    let scope_x = x + 6;
    let scope_y = y + 6;
    let scope_w = w - 12;

    // Snapshot up to SAMPLE_BUDGET samples.
    let n = vec.snapshot(SAMPLE_BUDGET, vec_l, vec_r);
    let mode = VectorMode::from_u32(params.vector_mode.load(Ordering::Relaxed));

    let cx = scope_x + scope_w / 2;
    let cy = scope_y + scope_h / 2;
    let r = scope_w.min(scope_h) / 2 - 6;

    // Background grid (a faint cross + diagonals).
    let grid_color = theme::TEXT_DIM;
    for dy in 0..scope_h {
        if dy == scope_h / 2 {
            for dx in 0..scope_w {
                draw_rect(pixmap, scope_x + dx, scope_y + dy, 1, 1, grid_color);
            }
        }
    }
    for dx in 0..scope_w {
        if dx == scope_w / 2 {
            for dy in 0..scope_h {
                draw_rect(pixmap, scope_x + dx, scope_y + dy, 1, 1, grid_color);
            }
        }
    }

    match mode {
        VectorMode::Polar => draw_polar(pixmap, cx, cy, r, &vec_l[..n], &vec_r[..n]),
        VectorMode::Lissajous => draw_lissajous(pixmap, cx, cy, r, &vec_l[..n], &vec_r[..n]),
    }

    // Correlation bar at the bottom
    let corr = f32::from_bits(params.correlation.load(Ordering::Relaxed));
    let bar_y = y + h - 28;
    draw_meter_bar(pixmap, x + 6, bar_y, w - 12, 8, corr, theme::cyan_to_pink(if corr > 0.0 { 0.0 } else { 1.0 }));

    // Balance bar (-1 left, +1 right)
    let bal = f32::from_bits(params.balance.load(Ordering::Relaxed));
    let bal_y = y + h - 16;
    draw_meter_bar(pixmap, x + 6, bal_y, w - 12, 8, bal, theme::ACCENT);
}

/// Polar-sample dot cloud. Each (L, R) sample is drawn as a single pixel rotated 45°
/// (so mono = vertical axis, anti-correlated = horizontal axis). Pink for L-leaning,
/// cyan for R-leaning.
fn draw_polar(pixmap: &mut PixmapMut<'_>, cx: i32, cy: i32, radius: i32, l: &[f32], r: &[f32]) {
    let r_f = radius as f32;
    let inv_sqrt2 = std::f32::consts::FRAC_1_SQRT_2;
    for i in 0..l.len() {
        // Rotate 45°: x = (L - R) / √2, y = -(L + R) / √2 (negate y so up = mono).
        let xn = (l[i] - r[i]) * inv_sqrt2;
        let yn = -((l[i] + r[i]) * inv_sqrt2);
        let px = cx + (xn.clamp(-1.0, 1.0) * r_f) as i32;
        let py = cy + (yn.clamp(-1.0, 1.0) * r_f) as i32;
        // Color by L-vs-R dominance.
        let bias = if l[i].abs() > r[i].abs() {
            ((l[i].abs() - r[i].abs()) / (l[i].abs() + r[i].abs() + 1e-9)).clamp(0.0, 1.0)
        } else {
            -((r[i].abs() - l[i].abs()) / (l[i].abs() + r[i].abs() + 1e-9)).clamp(0.0, 1.0)
        };
        // bias > 0 → pink (L), bias < 0 → cyan (R)
        let t = (bias + 1.0) * 0.5;
        let color = theme::cyan_to_pink(t);
        draw_rect(pixmap, px, py, 1, 1, color);
    }
}

/// Lissajous: L on X, R on Y, no rotation.
fn draw_lissajous(pixmap: &mut PixmapMut<'_>, cx: i32, cy: i32, radius: i32, l: &[f32], r: &[f32]) {
    let r_f = radius as f32;
    for i in 0..l.len() {
        let px = cx + (l[i].clamp(-1.0, 1.0) * r_f) as i32;
        let py = cy - (r[i].clamp(-1.0, 1.0) * r_f) as i32;
        draw_rect(pixmap, px, py, 1, 1, theme::ACCENT);
    }
}

/// Draws a horizontal bar with the marker positioned according to `value` ∈ [-1, 1].
fn draw_meter_bar(pixmap: &mut PixmapMut<'_>, x: i32, y: i32, w: i32, h: i32, value: f32, color: Color) {
    draw_rect(pixmap, x, y, w, h, theme::PANEL_BG);
    draw_rect_outline(pixmap, x, y, w, h, theme::BORDER);
    // Center line
    let cx = x + w / 2;
    draw_rect(pixmap, cx, y + 1, 1, h - 2, theme::TEXT_DIM);
    // Marker
    let v = value.clamp(-1.0, 1.0);
    let marker_x = x + ((v + 1.0) * 0.5 * w as f32) as i32;
    draw_rect(pixmap, marker_x - 1, y + 1, 3, h - 2, color);
}
```

```rust
pub fn compute_layout(x: i32, y: i32, w: i32, h: i32) -> BandStripLayout {
    let band_w = w / 4;
    let band_x = [x, x + band_w, x + 2 * band_w, x + 3 * band_w];

    // Inside a band: Width slider on left half (vertical, narrow), knob + buttons on right.
    let pad = 6;
    let width_rect = (pad, pad, 24, h - 2 * pad);
    let stz_center = (band_w / 2 + 12, h / 2 - 16);
    let stz_radius = (band_w / 4).min(h / 4) - 4;
    let mode_rect = (band_w - 38, h / 2 + 8, 32, 14);
    let solo_rect = (band_w - 38, h / 2 + 26, 32, 14);

    BandStripLayout { band_x, band_w, y, h, width_rect, stz_center, stz_radius, mode_rect, solo_rect }
}

pub fn draw(

        params.b2_mode.value(),
        params.b3_mode.value(),
    ];

        params.b1_solo.value(),
        params.b2_solo.value(),

        // Width slider (vertical bar with marker)

        draw_rect(pixmap, slot_x, slot_y, ww, wh, theme::SPECTRUM_BG);

        draw_rect(pixmap, slot_x, marker_y - 1, ww, 3, theme::cyan_to_pink(w_norm));

        // Stereoize knob (simple arc representation: ring + amount segment)
        let (cx, cy) = (bx + layout.stz_center.0, layout.y + layout.stz_center.1);
        let radius = layout.stz_radius;
        // Ring background
        draw_arc_ring(pixmap, cx, cy, radius, 0.0, 1.0, theme::SPECTRUM_BG);
        // Amount arc
        let stz_norm = (stz_amounts[i] / 100.0).clamp(0.0, 1.0);
        if stz_norm > 0.0 {
            draw_arc_ring(pixmap, cx, cy, radius, 0.0, stz_norm, theme::ACCENT);
        }

        // Mode toggle
        let (mx, my, mw, mh) = layout.mode_rect;
        let mode_x = bx + mx;
        let mode_y = layout.y + my;
        draw_rect(pixmap, mode_x, mode_y, mw, mh, theme::SPECTRUM_BG);
        draw_rect_outline(pixmap, mode_x, mode_y, mw, mh, theme::BORDER);
        // Highlight active half (Mode I = left, Mode II = right)
        let half_w = mw / 2;
        let active_x = if matches!(modes[i], crate::StereoizeModeParam::I) { mode_x } else { mode_x + half_w };
        draw_rect(pixmap, active_x, mode_y, half_w, mh, theme::ACCENT);

        // Solo button

        let solo_x = bx + sx;
        let solo_y = layout.y + sy;
        let color = if solos[i] { theme::cyan_to_pink(0.5) } else { theme::SPECTRUM_BG };
        draw_rect(pixmap, solo_x, solo_y, sw, sh, color);
        draw_rect_outline(pixmap, solo_x, solo_y, sw, sh, theme::BORDER);
    }

/// Draw a partial ring from `start_norm` to `end_norm` (both in [0, 1]),
/// where 0 = top of ring (12 o'clock) and progress is clockwise.
/// Implemented as a series of small filled rectangles around the perimeter.
fn draw_arc_ring(pixmap: &mut PixmapMut<'_>, cx: i32, cy: i32, radius: i32, start_norm: f32, end_norm: f32, color: tiny_skia::Color) {
    let r_inner = (radius - 3).max(1) as f32;
    let r_outer = radius as f32;
    let steps = 60;
    let start = start_norm * std::f32::consts::TAU;
    let end = end_norm * std::f32::consts::TAU;
    if end <= start { return; }
    for i in 0..steps {
        let t = i as f32 / steps as f32;
        let angle = start + (end - start) * t;
        // 12 o'clock origin: angle 0 = -π/2
        let real_angle = angle - std::f32::consts::FRAC_PI_2;
        let r_mid = (r_inner + r_outer) * 0.5;
        let px = cx + (real_angle.cos() * r_mid) as i32;
        let py = cy + (real_angle.sin() * r_mid) as i32;
        draw_rect(pixmap, px - 1, py - 1, 3, 3, color);
    }
}
```

```rust
    /// Width drag with Link Bands behavior. `requested_norm` is the normalized value the
    /// dragged band would receive; if Link is on, we compute the delta vs that band's
    /// drag-start baseline and clamp it to the smallest available headroom.
    fn maybe_apply_link_width(&self, dragged: usize, requested_norm: f32, started: bool, ended: bool, dragged_p: &nih_plug::params::FloatParam) {
        if !self.params.link_bands.value() {
            // Independent drag.
            let setter = nih_plug::prelude::ParamSetter::new(self.ctx.as_ref());
            if started { setter.begin_set_parameter(dragged_p); }
            setter.set_parameter_normalized(dragged_p, requested_norm);
            if ended { setter.end_set_parameter(dragged_p); }
            return;
        }

        let requested_value = -100.0 + requested_norm * 200.0; // de-normalize Width
        let dragged_baseline = self.link_baseline.width_at_drag_start[dragged];
        let requested_delta = requested_value - dragged_baseline;

        // Clamp delta against each band's headroom.
        let mut allowed_delta = requested_delta;
        let params = self.width_params();
        for i in 0..4 {
            let baseline = self.link_baseline.width_at_drag_start[i];
            let target = baseline + allowed_delta;
            if target > 100.0 {
                allowed_delta = allowed_delta.min(100.0 - baseline);
            }
            if target < -100.0 {
                allowed_delta = allowed_delta.max(-100.0 - baseline);
            }
        }

        for i in 0..4 {
            let p = params[i];
            let target = (self.link_baseline.width_at_drag_start[i] + allowed_delta).clamp(-100.0, 100.0);
            let new_norm = (target + 100.0) / 200.0;
            let setter = nih_plug::prelude::ParamSetter::new(self.ctx.as_ref());
            if started { setter.begin_set_parameter(p); }
            setter.set_parameter_normalized(p, new_norm);
            if ended { setter.end_set_parameter(p); }
        }
    }

    fn maybe_apply_link_stz(&self, dragged: usize, requested_norm: f32, started: bool, ended: bool, dragged_p: &nih_plug::params::FloatParam) {
        if !self.params.link_bands.value() {
            let setter = nih_plug::prelude::ParamSetter::new(self.ctx.as_ref());
            if started { setter.begin_set_parameter(dragged_p); }
            setter.set_parameter_normalized(dragged_p, requested_norm);
            if ended { setter.end_set_parameter(dragged_p); }
            return;
        }
        let requested_value = requested_norm * 100.0;
        let dragged_baseline = self.link_baseline.stz_at_drag_start[dragged];
        let requested_delta = requested_value - dragged_baseline;
        let mut allowed_delta = requested_delta;
        for i in 0..4 {
            let baseline = self.link_baseline.stz_at_drag_start[i];
            let target = baseline + allowed_delta;
            if target > 100.0 { allowed_delta = allowed_delta.min(100.0 - baseline); }
            if target < 0.0   { allowed_delta = allowed_delta.max(-baseline); }
        }
        let params = self.stz_params();
        for i in 0..4 {
            let p = params[i];
            let target = (self.link_baseline.stz_at_drag_start[i] + allowed_delta).clamp(0.0, 100.0);
            let new_norm = target / 100.0;
            let setter = nih_plug::prelude::ParamSetter::new(self.ctx.as_ref());
            if started { setter.begin_set_parameter(p); }
            setter.set_parameter_normalized(p, new_norm);
            if ended { setter.end_set_parameter(p); }
        }
    }

    fn snapshot_link_baseline(&mut self) {
        for i in 0..4 {
            self.link_baseline.width_at_drag_start[i] = self.width_params()[i].value();
            self.link_baseline.stz_at_drag_start[i] = self.stz_params()[i].value();
        }
    }
}
```

```rust
    fn on_event(&mut self, _window: &mut Window, event: Event) -> EventStatus {
        match event {
            Event::Mouse(MouseEvent::CursorMoved { position, .. }) => {
                self.mouse_x = position.x as f32;
                self.mouse_y = position.y as f32;

                if !matches!(self.drag_target, DragTarget::None) {
                    self.update_drag(self.mouse_x, self.mouse_y);
                    return EventStatus::Captured;
                }
                EventStatus::Ignored
            }
            Event::Mouse(MouseEvent::ButtonPressed { button, .. }) => {

                match button {

                        self.handle_right_click(target);
                        EventStatus::Captured
                    }
                    _ => EventStatus::Ignored,
                }
            }
            Event::Mouse(MouseEvent::ButtonReleased { button, .. }) => {
                if matches!(button, MouseButton::Left) && !matches!(self.drag_target, DragTarget::None) {
                    self.update_drag_with_phase(self.mouse_x, self.mouse_y, false, true);
                    self.drag_target = DragTarget::None;
                }
                EventStatus::Captured
            }
            Event::Window(WindowEvent::Resized(info)) => {
                let (w, h) = (info.physical_size.width as u32, info.physical_size.height as u32);
                let packed = ((w as u64) << 32) | (h as u64);
                self.pending_resize.store(packed, Ordering::Release);
                EventStatus::Captured
            }
            Event::Keyboard(kbd) => {

impl ImagineWindow {
    fn update_drag(&mut self, mx: f32, my: f32) {
        self.update_drag_with_phase(mx, my, false, false);
    }

    fn update_drag_with_phase(&mut self, mx: f32, my: f32, started: bool, ended: bool) {
        let target = self.drag_target;
        match target {
            DragTarget::Split { idx } => {
                let spec_x = self.right_x() as f32;
                let spec_w = self.right_w() as f32;
                let x_norm = ((mx - spec_x) / spec_w).clamp(0.0, 1.0);
                let hz = x_to_hz(x_norm);
                // Convert hz to normalized param value via the Skewed range.
                let p = match idx {
                    0 => &self.params.xover_1,
                    1 => &self.params.xover_2,
                    _ => &self.params.xover_3,
                };
                let normalized = p.preview_normalized(hz);
                self.set_param_normalized(target, normalized, started, ended);
            }
            DragTarget::BandWidth { band: _ } => {
                let layout = compute_band_layout(self.right_x(), self.band_y(), self.right_w(), self.band_h());
                let (_, wy, _, wh) = layout.width_rect;
                let local_y = (my as i32 - layout.y - wy).clamp(0, wh);
                // Top = +100 (norm 1.0), bottom = -100 (norm 0.0)
                let normalized = 1.0 - local_y as f32 / wh as f32;
                self.set_param_normalized(target, normalized, started, ended);
            }
            DragTarget::BandStz { band: _ } => {
                // Vertical drag changes amount: top of knob area = 100, bottom = 0.
                // Use anchored delta-y for finer control.
                let dy = self.drag_anchor_y - my;
                let pixels_per_full = 200.0;
                let delta_norm = (dy / pixels_per_full).clamp(-1.0, 1.0);
                // Use the band's drag-start baseline as the anchor.
                let band_idx = if let DragTarget::BandStz { band } = target { band } else { return; };
                let baseline_norm = self.link_baseline.stz_at_drag_start[band_idx] / 100.0;
                let normalized = (baseline_norm + delta_norm).clamp(0.0, 1.0);
                self.set_param_normalized(target, normalized, started, ended);
            }
            DragTarget::Recover => {
                // Horizontal drag across the recover_rect.
                let g = compute_global_layout(0, self.main_h(), self.width as i32, self.global_h());
                let (rx, _, rw, _) = g.recover_rect;
                let normalized = ((mx - rx as f32) / rw as f32).clamp(0.0, 1.0);
                self.set_param_normalized(target, normalized, started, ended);
            }
            DragTarget::None => {}
        }
    }

    fn handle_discrete_click(&mut self, mx: f32, my: f32) {
        let mxi = mx as i32;
        let myi = my as i32;

        // Band strip: Mode toggle / Solo button hits.
        if myi >= self.band_y() && myi < self.band_y() + self.band_h() {
            let layout = compute_band_layout(self.right_x(), self.band_y(), self.right_w(), self.band_h());
            for i in 0..4 {
                let bx = layout.band_x[i];
                if mxi < bx || mxi >= bx + layout.band_w { continue; }
                let local_x = mxi - bx;
                let local_y = myi - layout.y;

                // Mode toggle (2 halves)
                let (mxi_, myi_, mw, mh) = layout.mode_rect;
                if local_x >= mxi_ && local_x < mxi_ + mw && local_y >= myi_ && local_y < myi_ + mh {
                    let half_w = mw / 2;
                    let new_mode = if local_x - mxi_ < half_w {
                        crate::StereoizeModeParam::I
                    } else {
                        crate::StereoizeModeParam::Ii
                    };
                    self.set_mode(i, new_mode);
                    return;
                }

                // Solo button — radio behavior (turn off other solos).
                let (sxi, syi, sw, sh) = layout.solo_rect;
                if local_x >= sxi && local_x < sxi + sw && local_y >= syi && local_y < syi + sh {
                    self.toggle_solo_radio(i);
                    return;
                }
            }
        }

        // Global strip: Link Bands toggle, Quality selector
        if myi >= self.main_h() && myi < self.height as i32 {
            let g = compute_global_layout(0, self.main_h(), self.width as i32, self.global_h());
            // Link
            let (lx, ly, lw, lh) = g.link_rect;
            if mxi >= lx && mxi < lx + lw && myi >= ly && myi < ly + lh {
                let p = &self.params.link_bands;
                let new_val = !p.value();
                let setter = nih_plug::prelude::ParamSetter::new(self.ctx.as_ref());
                setter.begin_set_parameter(p);
                setter.set_parameter(p, new_val);
                setter.end_set_parameter(p);
                return;
            }
            // Quality (2 halves)
            let (qx, qy, qw, qh) = g.quality_rect;
            if mxi >= qx && mxi < qx + qw && myi >= qy && myi < qy + qh {
                let half_w = qw / 2;
                let new_q = if mxi - qx < half_w { Quality::Linear } else { Quality::Iir };
                let p = &self.params.quality;
                let norm = match new_q {
                    Quality::Linear => 0.0,
                    Quality::Iir => 1.0,
                };
                let setter = nih_plug::prelude::ParamSetter::new(self.ctx.as_ref());
                setter.begin_set_parameter(p);
                setter.set_parameter_normalized(p, norm);
                setter.end_set_parameter(p);
                return;
            }
        }

        // Vectorscope mode toggle (small region at top of vec panel — see Task 14 layout).
        // For simplicity, click on bottom-left of vec panel toggles polar/lissajous.
        if mxi < self.vec_w() && myi < 30 {
            let cur = self.params.vector_mode.load(Ordering::Relaxed);
            let new = if cur == 0 { 1 } else { 0 };
            self.params.vector_mode.store(new, Ordering::Release);
        }
    }

    fn set_mode(&self, band: usize, mode: crate::StereoizeModeParam) {
        let p: &nih_plug::params::EnumParam<crate::StereoizeModeParam> = match band {
            0 => &self.params.b0_mode,
            1 => &self.params.b1_mode,
            2 => &self.params.b2_mode,
            _ => &self.params.b3_mode,
        };
        let norm = match mode {
            crate::StereoizeModeParam::I => 0.0,
            crate::StereoizeModeParam::Ii => 1.0,

        let setter = nih_plug::prelude::ParamSetter::new(self.ctx.as_ref());

        setter.set_parameter_normalized(p, norm);
        setter.end_set_parameter(p);

    fn toggle_solo_radio(&self, clicked: usize) {
        let solos = [&self.params.b0_solo, &self.params.b1_solo, &self.params.b2_solo, &self.params.b3_solo];
        let was_active = solos[clicked].value();
        for (i, p) in solos.iter().enumerate() {
            let new_val = if i == clicked { !was_active } else { false };
            let cur = p.value();
            if cur == new_val { continue; }
            let setter = nih_plug::prelude::ParamSetter::new(self.ctx.as_ref());
            setter.begin_set_parameter(p);
            setter.set_parameter(p, new_val);
            setter.end_set_parameter(p);
        }

    fn handle_right_click(&mut self, target: DragTarget) {
        // Stepped/toggle controls are no-op on right-click. Continuous targets open text-edit.
        match target {
            DragTarget::Split { .. } | DragTarget::BandWidth { .. } |
            DragTarget::BandStz { .. } | DragTarget::Recover => {

                let initial = self.current_value_string(target);
                self.text_edit.begin(target, &initial);
            }
            DragTarget::None => {}
        }
    }

        match target {

                let p = match idx {
                    0 => &self.params.xover_1,
                    1 => &self.params.xover_2,
                    _ => &self.params.xover_3,
                };
                p.normalized_value_to_string(p.modulated_normalized_value(), false)
            }
            DragTarget::BandWidth { band } => {
                let p: &nih_plug::params::FloatParam = match band {
                    0 => &self.params.b0_width, 1 => &self.params.b1_width,
                    2 => &self.params.b2_width, _ => &self.params.b3_width,
                };
                p.normalized_value_to_string(p.modulated_normalized_value(), false)

            DragTarget::BandStz { band } => {

                    0 => &self.params.b0_stz, 1 => &self.params.b1_stz,
                    2 => &self.params.b2_stz, _ => &self.params.b3_stz,
                };
                p.normalized_value_to_string(p.modulated_normalized_value(), false)

            DragTarget::Recover => {
                let p = &self.params.recover_sides;
                p.normalized_value_to_string(p.modulated_normalized_value(), false)

            DragTarget::None => String::new(),
        }

    fn handle_text_edit_commit(&mut self) {
        // Drive the TextEditState from keyboard events (call from on_event match for Keyboard).
        // When the user presses Enter, call self.text_edit.commit() which returns
        // Some((target, text)) on commit; map keys to insert_char / backspace / cancel
        // following pope-scope's text-entry pattern.
        if let Some((target, committed_text)) = self.text_edit.commit() {
            // Try to parse via the param's string_to_normalized_value.
            let normalized_opt = match target {
                DragTarget::Split { idx } => {
                    let p = match idx {
                        0 => &self.params.xover_1,
                        1 => &self.params.xover_2,
                        _ => &self.params.xover_3,
                    };
                    p.string_to_normalized_value(&committed_text)
                }
                DragTarget::BandWidth { band } => {
                    let p: &nih_plug::params::FloatParam = match band {
                        0 => &self.params.b0_width, 1 => &self.params.b1_width,
                        2 => &self.params.b2_width, _ => &self.params.b3_width,
                    };
                    p.string_to_normalized_value(&committed_text)
                }
                DragTarget::BandStz { band } => {
                    let p: &nih_plug::params::FloatParam = match band {
                        0 => &self.params.b0_stz, 1 => &self.params.b1_stz,
                        2 => &self.params.b2_stz, _ => &self.params.b3_stz,
                    };
                    p.string_to_normalized_value(&committed_text)
                }
                DragTarget::Recover => {
                    self.params.recover_sides.string_to_normalized_value(&committed_text)
                }
                DragTarget::None => None,
            };
            if let Some(normalized) = normalized_opt {
                self.set_param_normalized(target, normalized, true, true);
            }
        }
    }
```
